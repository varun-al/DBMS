CREATE TABLE EMPLOYEE(
EID INT PRIMARY KEY,
NAME VARCHAR2(20),
ADDRESS VARCHAR2(20),
GENDER CHAR(1) CHECK(GENDER='M' OR GENDER='F'),
SALRY NUMBER(6),
SUPEREID REFERENCES EMPLOYEE(EID),
DNO NUMBER);
